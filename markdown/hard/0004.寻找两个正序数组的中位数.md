#### C

```
double findMedianSortedArrays(int* nums1, int nums1Size, int* nums2, int nums2Size) {
    int max = nums1Size + nums2Size;
    int arr[max];
    int i = 0, j = 0, k = 0;
    while (k < max) {
        if (i < nums1Size && j < nums2Size) {
            arr[k++] = nums1[i] < nums2[j] ? nums1[i++] : nums2[j++];
        } else if (i >= nums1Size) {
            arr[k++] = nums2[j++];
        } else if (j >= nums2Size) {
            arr[k++] = nums1[i++];
        }
    }
    double res;
    if (max % 2 == 0) {
        res = (arr[max / 2 - 1] + arr[max / 2]) / 2.0;
    } else {
        res = arr[max / 2];
    }
    return res;
}
```
