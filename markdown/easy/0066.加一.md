#### C++

```
class Solution {
public:
    vector<int> plusOne(vector<int>& digits) {
        for (int i = digits.size() - 1; i >= 0; i--) {
            digits[i] = (digits[i] + 1) % 10;
            if (digits[i] != 0) {
                return digits;
            }
        }
        vector<int> result(digits.size() + 1);
        result[0] = 1;
        return result;
    }
};
```
