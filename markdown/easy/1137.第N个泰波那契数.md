#### C++

```
class Solution {
public:
    int tribonacci(int n) {
        if (n == 0) {
            return 0;
        }
        if (n < 3) {
            return 1;
        }
        int x = 0;
        int y = 1;
        int z = 1;
        int tri = -1;
        for (int i = 3; i <= n; i++) {
            tri = x + y + z;
            x = y;
            y = z;
            z = tri;
        }
        return tri;
    }
};
```
