class Solution {
    final int MOD = 1000000007;

    public int fib(int n) {
        if (n < 2) {
            return n;
        }
        int x = 0;
        int y = 1;
        int fib = -1;
        for (int i = 2; i <= n; i++) {
            fib = (x + y) % MOD;
            x = y;
            y = fib;
        }
        return fib;
    }
}
