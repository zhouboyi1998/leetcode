#### Java

```
class Solution {
    public boolean isUnique(String astr) {
        for (int i = 0; i < astr.length() - 1; i++) {
            if (astr.substring(i + 1).indexOf(astr.charAt(i)) != -1) {
                return false;
            }
        }
        return true;
    }
}
```

#### Go

func isUnique(astr string) bool {
    len := len(astr);
    for i := 0; i < len - 1; i++ {
        if strings.Index(astr[i + 1 : len], string(astr[i])) != -1 {
            return false;
        }
    }
    return true;
}
