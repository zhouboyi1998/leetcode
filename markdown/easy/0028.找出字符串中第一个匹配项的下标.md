class Solution {
    public int strStr(String haystack, String needle) {
        int length1 = haystack.length();
        int length2 = needle.length();
        for (int i = 0; i < length1 - length2 + 1; i++) {
            int length = 0;
            for (int j = 0; j < length2; j++) {
                if (haystack.charAt(i + j) == needle.charAt(j)) {
                    length++;
                } else {
                    break;
                }
            }
            if (length == length2) {
                return i;
            }
        }
        return -1;
    }
}
