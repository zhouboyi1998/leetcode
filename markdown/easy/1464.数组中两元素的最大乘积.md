#### Java

```
class Solution {
    public int maxProduct(int[] nums) {
        int f = 0;
        int s = 0;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] > f) {
                s = f;
                f = nums[i];
            } else if (nums[i] > s) {
                s = nums[i];
            }
        }
        return (f - 1) * (s - 1);
    }
}
```

impl Solution {
    pub fn max_product(nums: Vec<i32>) -> i32 {
        let mut f: i32 = 0;
        let mut s: i32 = 0;
        for x in nums.iter() {
            if *x > f {
                s = f;
                f = *x;
            } else if *x > s {
                s = *x;
            }
        }
        return (f - 1) * (s - 1);
    }
}
